10.2 Upišite sljedeći program   ( TINKERCAD NEĆE POKAZIVATI DRUGU DECIMALU )

#include <LiquidCrystal.h>
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);
void setup() {
  lcd.begin(16, 2);
  lcd.setCursor(0,0);
  lcd.print("   VOLTMETAR  ");
  lcd.setCursor(10,1);           
  lcd.print("V"); 
}
void loop() {
  float napon = 0.004887 * analogRead(A0); 
  lcd.setCursor(5,1);           
  lcd.print(napon); 
  delay(30);

}
