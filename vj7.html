7.1  Izrada elektroničke kocke

Električna shema

Montažna shema

Upišite sljedeći program

int tipka;

int broj=0;

int pauza = 300;

 

void setup() {

  pinMode(2,INPUT_PULLUP);

  pinMode(3,OUTPUT);

  pinMode(4,OUTPUT);

  pinMode(5,OUTPUT);

  pinMode(6,OUTPUT);

  pinMode(7,OUTPUT);

  pinMode(8,OUTPUT);

}

 

void loop() {

  tipka=digitalRead(2);

  

  if(tipka==0) {

    digitalWrite(3,HIGH);

    delay(pauza);

    digitalWrite(3,LOW);

    digitalWrite(4,HIGH);

    delay(pauza);

    digitalWrite(4,LOW);

    digitalWrite(5,HIGH);

    delay(pauza);

    digitalWrite(5,LOW);

    digitalWrite(8,HIGH);

    delay(pauza);

    digitalWrite(8,LOW);

    digitalWrite(7,HIGH);

    delay(pauza);

    digitalWrite(7,LOW);

    digitalWrite(6,HIGH);

    delay(pauza);

    digitalWrite(6,LOW);

    

    broj=random(1,7);

  }

 

  if(tipka==1) {        

 

      if(broj==1) {

        digitalWrite(3,HIGH);

        digitalWrite(4,LOW);

        digitalWrite(5,LOW);

        digitalWrite(6,LOW);

        digitalWrite(7,LOW);

        digitalWrite(8,LOW);

        }

  

 

       

        if(broj==3) {

        digitalWrite(3,HIGH);

        digitalWrite(4,LOW);

        digitalWrite(5,LOW);

        digitalWrite(6,HIGH);

        digitalWrite(7,LOW);

        digitalWrite(8,HIGH);

        }

  

        if(broj==4) {

        digitalWrite(3,HIGH);

        digitalWrite(4,LOW);

        digitalWrite(5,HIGH);

        digitalWrite(6,HIGH);

        digitalWrite(7,LOW);

        digitalWrite(8,HIGH);

        }

  

        if(broj==5) {

        digitalWrite(3,HIGH);

        digitalWrite(4,HIGH);

        digitalWrite(5,HIGH);

        digitalWrite(6,HIGH);

        digitalWrite(7,LOW);

        digitalWrite(8,HIGH);

        }

  

        if(broj==6) {

        digitalWrite(3,HIGH);

        digitalWrite(4,HIGH);

        digitalWrite(5,HIGH);

        digitalWrite(6,HIGH);

        digitalWrite(7,HIGH);

        digitalWrite(8,HIGH);

        }

    }

}
